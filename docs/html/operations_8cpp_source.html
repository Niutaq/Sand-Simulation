150sf::Color getRandomShadeAlpha(sf::Color baseColor) {
151 std::random_device rd;
152
153 std::mt19937 gen(rd());
154 std::uniform_real_distribution<float> distribution(0.9, 1.0);
155 sf::Color shadeColor;
156 shadeColor.r = static_cast<sf::Uint8>(baseColor.r * distribution(gen));
157 shadeColor.g = static_cast<sf::Uint8>(baseColor.g * distribution(gen));
158 shadeColor.b = static_cast<sf::Uint8>(baseColor.b * distribution(gen));
159
161 shadeColor.a = 255;
162
163 return shadeColor;
